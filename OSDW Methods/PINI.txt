# Add OS Initialisations into PINI method.
#
# Note that in some cases your DSDT doesn't contain a PINI method, particulary on the 7 Series
# 
# motherboards and earlier. So it needs to be added in the _INI Method, on the Scope _SB.PCI0
# 
# where the OSYS and _OSI are. Then, be carefull on what you do, otherwise your system 
# 
# will not boot at all into the OS, or it will keep crashing on Windows and KP on macOS.
# 
# Well, now will only write, Darwin, Linux, W7, W10

# Remove existing PINI method to add new one in the Scope (\)
into method label PINI remove_entry end;

# Insert PINI Method into Scope \
into scope label \ code_regex PICM insert
begin
Method (PINI, 0, NotSerialized)\n
{\n
    Store (0x07D9, OSYS)\n
    If (CondRefOf (_OSI, Local0))\n
    {\n
        If (_OSI ("Darwin"))\n
        {\n
            Store (0x2710, OSYS)\n
        }\n
        Else\n
        {\n
            If (_OSI ("linux"))\n
            {\n
                Store (0x03E8, OSYS)\n
            }\n
            Else\n
            {\n
                If (_OSI ("Windows 2009"))\n
                {\n
                    Store (0x07D9, OSYS)\n
                }\n
                Else\n
                {\n
                    If (_OSI ("Windows 2012"))\n
                    {\n
                        Store (0x07DC, OSYS)\n
                    }\n
                    Else\n
                    {\n
                        If (_OSI ("Windows 2013"))\n
                        {\n
                            Store (0x07DD, OSYS)\n
                        }\n
                        Else\n
                        {\n
                            If (_OSI ("Windows 2015"))\n
                            {\n
                                Store (0x07DF, OSYS)\n
                            }\n
                        }\n
                    }\n
                }\n
            }\n
        }\n
    }\n
    Else\n
    {\n
        Store (0x07D9, OSYS)\n
    }\n
}\n
end;

# Add PINI method into the scope _SB and remove it from the definitionblock
into method label PINI remove_entry;
into definitionblock code_regex . insert
begin
Scope (_SB)\n
{\n
    Method (PINI, 0, NotSerialized)  // _INI: Initialize\n
    {\n
    }\n
}\n
end;
